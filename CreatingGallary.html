const floorSize = 10; // 床のサイズを保持（10x10） let isClick = false; let clickStartTime = 0; // マウス押下時：クリック判定の準備 window.addEventListener('mousedown', () => { isClick = true; clickStartTime = performance.now(); }); // マウス移動でドラッグと判定 window.addEventListener('mousemove', () => { if (performance.now() - clickStartTime > 200) { isClick = false; } }); // マウスアップ時にクリックと判定された場合だけ実行 window.addEventListener('mouseup', (event) => { if (!isClick) return; mouse.x = (event.clientX / window.innerWidth) * 2 - 1; mouse.y = -(event.clientY / window.innerHeight) * 2 + 1; raycaster.setFromCamera(mouse, camera); const intersects = raycaster.intersectObject(floor); if (intersects.length > 0) { const lookAtPos = intersects[0].point; // 範囲外チェック（床より大幅に外れたら無視） const limit = floorSize / 2 + 0.5; if (Math.abs(lookAtPos.x) > limit || Math.abs(lookAtPos.z) > limit) return; // 視線方向：LookAt位置 → カメラ位置 const direction = new THREE.Vector3(); direction.subVectors(lookAtPos, camera.position).normalize(); // カメラ位置を LookAt から1m手前に設定（高さは固定） const newCamPos = lookAtPos.clone().addScaledVector(direction, -1); newCamPos.y = 2.5; // アニメーション準備 moveStart = performance.now() / 1000; moveFrom.copy(camera.position); moveTo.copy(newCamPos); // ターゲット更新（次フレームで反映） controls.target.copy(lookAtPos); } });
